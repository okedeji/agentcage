1. Store connection secrets in Vault:

{{- if .Values.postgres.enabled }}
   # Get the CNPG-generated password:
   kubectl get secret {{ include "agentcage-infra.fullname" . }}-postgres-app \
     -o jsonpath='{.data.password}' | base64 -d

   agentcage vault put orchestrator postgres-url \
     "postgres://{{ .Values.postgres.owner }}:<password>@{{ include "agentcage-infra.fullname" . }}-postgres-rw:5432/{{ .Values.postgres.database }}"
{{- end }}

{{- if .Values.nats.enabled }}
   agentcage vault put orchestrator nats-url \
     "nats://{{ .Release.Name }}-nats:4222"
{{- end }}

{{- if .Values.nomad.enabled }}
   # Bootstrap Nomad ACL (one-time, requires nomad CLI):
   kubectl port-forward svc/{{ include "agentcage-infra.fullname" . }}-nomad {{ .Values.nomad.port }}:{{ .Values.nomad.port }} &
   NOMAD_ADDR=http://localhost:{{ .Values.nomad.port }} nomad acl bootstrap
   # Copy the Secret ID from the output, then:
   agentcage vault put orchestrator nomad-token "<Secret ID>"
{{- end }}

{{- end }}
